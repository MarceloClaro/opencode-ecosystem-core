# TC-01: Verificar não-redundância
classes_por_celula = {}
for i, cell in enumerate(nb["cells"]):
    if cell["cell_type"] == "code":
        classes = re.findall(r'class\s+(\w+)\s*[:\(]', "".join(cell["source"]))
        for c in classes:
            assert c not in classes_por_celula, f"{c} aparece nas cells {classes_por_celula[c]} e {i}"
            classes_por_celula[c] = i

# TC-02: Verificar imports em cada célula
for i, cell in enumerate(nb["cells"]):
    if cell["cell_type"] == "code":
        assert "import" in "".join(cell["source"]), f"Cell {i} sem imports"

# TC-03: Verificar consistência markdown→code
for i, cell in enumerate(nb["cells"]):
    if cell["cell_type"] == "markdown" and "FASE" in "".join(cell["source"]):
        assert i+1 < len(nb["cells"]), f"FASE sem code após cell {i}"
        assert nb["cells"][i+1]["cell_type"] == "code", f"Cell {i+1} não é code"
