% Redefine \texttt para permitir quebra em hífens no modo typewriter.
% A fonte Courier normalmente não hifeniza (hyphenchar=-1).
% Ativa quebra no '-' (ASCII 45) para evitar overfulls em paths técnicos.
% Preserva o funcionamento em math mode (onde \ttfamily é inválido).
\let\origtexttt\texttt
\protected\def\texttt#1{%
  \ifmmode
    \origtexttt{#1}%
  \else
    {\ttfamily\hyphenchar\font=45 \relax #1}%
  \fi
}
