\addcontentsline{toc}{section}{Manhã}

\begin{poema}
Texto ilustrativo --- substitua por um poema real.\\
Cada verso em uma linha, separado por \textbackslash\textbackslash.\\[6pt]
Uma linha em branco separa as estrofes.
\end{poema}
